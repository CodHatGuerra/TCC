CREATE SCHEMA IF NOT EXISTS `VF_Dev` DEFAULT CHARACTER SET utf8 ;
USE `VF_Dev` ;

CREATE TABLE IF NOT EXISTS `Endereco` (
  `ID` INT NOT NULL AUTO_INCREMENT,
  `Cep` VARCHAR(8) NOT NULL,
  `Uf` VARCHAR(2) NOT NULL,
  `Localidade` VARCHAR(80) NOT NULL,
  `Bairro` VARCHAR(120) NOT NULL,
  `Logradouro` VARCHAR(120) NOT NULL,
  `Numero` VARCHAR(8) NOT NULL,
  PRIMARY KEY (`ID`)
)
ENGINE = InnoDB;


CREATE TABLE IF NOT EXISTS `Vacina` (
  `ID` INT NOT NULL,
  `Nome` VARCHAR(150) NOT NULL,
  `Fábrica` VARCHAR(150) NOT NULL,
  `Data_Fabricacao` DATE NOT NULL,
  `Lote` VARCHAR(20) NOT NULL,
  `Data_Aplicacao` DATE NOT NULL,
  `Dose` VARCHAR(45) NOT NULL,
  `Data_Vinculada` DATE NOT NULL,
  PRIMARY KEY (`ID`),
  UNIQUE INDEX `ID_Vacina_UNIQUE` (`ID` ASC) VISIBLE
)
ENGINE = InnoDB;


CREATE TABLE IF NOT EXISTS `Carteira_vacina` (
  `ID` INT NOT NULL AUTO_INCREMENT,
  `Data_Registrada` DATE NOT NULL,
  `Vacina_ID` INT NULL,
  PRIMARY KEY (`ID`),
  UNIQUE INDEX `ID_Carteira_UNIQUE` (`ID` ASC) VISIBLE,
  INDEX `fk_Carteira_vacina_Vacina1_idx` (`Vacina_ID` ASC) VISIBLE,
  CONSTRAINT `fk_Carteira_vacina_Vacina1`
    FOREIGN KEY (`Vacina_ID`)
    REFERENCES `Vacina` (`ID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION
)
ENGINE = InnoDB;

  `ID` INT NOT NULL AUTO_INCREMENT,
